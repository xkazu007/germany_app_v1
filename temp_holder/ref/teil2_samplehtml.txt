        function selectOption(questionId, optionKey) {
            // Update Data

            // Update UI

            // 2. Add 'selected' to clicked option
            const selectedEl = document.getElementById(`q${questionId}-opt-${optionKey}`);
            if (selectedEl) selectedEl.classList.add('selected');
        }
